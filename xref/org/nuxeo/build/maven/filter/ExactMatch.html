22  public class ExactMatch extends SegmentMatch {
23  
24      public String segment;
25  
26      public ExactMatch(String segment) {
27          this.segment = segment;
28      }
29  
30      @Override
31      public boolean match(String segmentToMatch) {
32          return this.segment.equals(segmentToMatch);
33      }
34  
35      @Override
36      public String toString() {
37          return getClass().toString() + " (" + segment + ")";
38      }
39  
40  }
