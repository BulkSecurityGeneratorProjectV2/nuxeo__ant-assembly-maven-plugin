33  public class TestVersions {
34  
35      private static final Logger log = new ConsoleLogger();
36  
37      @Test
38      public void testVersionString() throws Exception {
39          // Some various filenames found in Nuxeo server
40          String[] filenames = new String[] { "nuxeo-1.0-SNAPSHOT.jar",
41                  "nuxeo-1-SNAPSHOT.jar", "nuxeo-1.0.0-SNAPSHOT.jar",
42                  "bcmail-jdk15-1.45.jar", "geronimo-connector-2.2.1-NX1.jar",
43                  "gmbal-api-only-3.1.0-b001.jar",
44                  "hibernate-annotations-3.4.0.GA.jar", "java-cup-0.11a.jar",
45                  "jboss-el-1.0_02.CR2.jar", "jboss-seam-2.1.0.SP1.jar",
46                  "jena-2.6.4-NX.jar", "jersey-core-1.11-minimal.jar",
47                  "jmd-0.8.1-tomasol-3e60e36137.jar",
48                  "jodconverter-core-3.0-NX7.jar", "xpp3-1.1.4c-clean.jar",
49                  "xsom-20081112.jar" };
50          Version version100SNAP = new Version(1, 0, 0);
51          version100SNAP.setSnapshot(true);
52          Version[] versions = new Version[] { version100SNAP, version100SNAP,
53                  version100SNAP, new Version(1, 45),
54                  new Version(2, 2, 1, "NX1"), new Version(3, 1, 0, "b001"),
55                  new Version(3, 4, 0, "GA"), new Version(0, 11, 0, "a"),
56                  new Version(1, 0, 0, "_02.CR2"), new Version(2, 1, 0, "SP1"),
57                  new Version(2, 6, 4, "NX"), new Version(1, 11, 0, "minimal"),
58                  new Version(0, 8, 1, "tomasol-3e60e36137"),
59                  new Version(3, 0, 0, "NX7"), new Version(1, 1, 4, "c-clean"),
60                  new Version(20081112) };
61          for (int i = 0; i < filenames.length; i++) {
62              Version version = Version.parse(getVersionFromFilename(filenames[i]));
63              assertEquals("Error parsing " + filenames[i], versions[i], version);
64          }
65      }
66  
67      private String getVersionFromFilename(String filename) {
68          Matcher m = RemoveDuplicateTask.PATTERN.matcher(filename);
69          assertTrue(m.find());
70          String key = filename.substring(0, m.start());
71          String v = m.group(1);
72          log.debug(filename + " => key=" + key + " version=" + v);
73          return v;
74      }
